\chapter{Definições de Símbolos}

\begin{itemize}
    \item $\mathbb{N}$ denota o conjunto dos números naturais;
    \item $\mathbb{Z}$ denota o conjunto dos números inteiros;
    \item $\mathbb{Q}$ denota o conjunto dos números racionais;
    \item $\mathbb{R}$ denota o conjunto dos números reais;
    \item $\mathbb{R}^+$ denota o conjunto dos números reais positivos;
    \item $|X|$ denota a cardinalidade do conjunto X;
    \item para $X \subseteq V$ denotamos por $\overline{X}$ o complemento de $X$ em $V$ , isto é, o conjunto $\frac{X}{V}$;
    \item $\binom{V}{2}$ denota o conjunto dos subconjuntos de $V$ de cardinalidade 2: $$ \binom{V}{2} = {X \subseteq V : |X| = 2} $$
    \item $\therefore$ representa \textit{tal que};
\end{itemize}
